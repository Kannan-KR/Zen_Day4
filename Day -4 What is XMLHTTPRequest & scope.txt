	b. var obj2 = { age:5, name: "Person 1" };

	var flag = "Match";

	for (var key in obj1) {
	  if (obj1[key] == obj2[key]) {
	    continue;
	  } else {
	    flag = "No match";
	    break;
	  }
	}
	console.log(flag);
